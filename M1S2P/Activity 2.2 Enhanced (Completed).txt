// Define the button and LED pin numbers
const uint8_t BTN_PIN = 2;
const uint8_t LED_PIN = 13;

// Store the previous state of the button and the current state of the LED
uint8_t buttonPrevState = LOW;
uint8_t ledState = LOW;

// Set up the initial configuration of the Arduino board
void setup()
{
  // Set the button pin as an input with a pull-up resistor enabled
  pinMode(BTN_PIN, INPUT_PULLUP);
  
  // Set the LED pin as an output
  pinMode(LED_PIN, OUTPUT);
  
  // Initialize the serial communication at a baud rate of 9600
  Serial.begin(9600);
  
  // Attach an interrupt to the button pin, triggering the 'interrupt()' function on CHANGE (button press or release)
  attachInterrupt(digitalPinToInterrupt(BTN_PIN), interrupt, CHANGE);
}

// The main loop of the Arduino program
void loop()
{
  // The loop is empty as we are using an interrupt to handle the button press
  // and toggle the LED state. So, there's no need for continuous checking here.
}

// The interrupt service triggered by the button press or release
void interrupt()
{
  // Toggle the state of the LED whenever the button state changes
  ledState = !ledState;
  digitalWrite(LED_PIN, ledState);
}
